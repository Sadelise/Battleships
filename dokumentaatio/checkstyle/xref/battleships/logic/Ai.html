3   import java.util.ArrayList;
4   import java.util.List;
5   import java.util.Random;
6   
7   public class Ai extends Player {
8   
9       private int[][] enemyMap;
10      private Boolean shipWasHit;
11      private List<Integer> destroyedX;
12      private List<Integer> destroyedY;
13      private List<Integer> acceptableX;
14      private List<Integer> acceptableY;
15      private Random guesser;
16  
17      public Ai() {
18          guesser = new Random();
19          ShipBuilder sb = new ShipBuilder(this, guesser);
20          sb.buildShips(6, 5);
21          enemyMap = new int[10][10];
22          shipWasHit = false;
23          destroyedX = new ArrayList<>();
24          destroyedY = new ArrayList<>();
25      }
26  
27      public int[] getShootingCoordinates() {
28          acceptableX = new ArrayList();
29          acceptableY = new ArrayList();
30          int[] coordinates = {-1, -1};
31          acceptableCoordinates();
32          if (acceptableX.size() > 0) {
33              int randomIndex = guesser.nextInt(acceptableX.size());
34              coordinates[0] = acceptableX.get(randomIndex);
35              coordinates[1] = acceptableY.get(randomIndex);
36          }
37          return coordinates;
38      }
39  
40      public void feedback(Boolean hit, Boolean sank, int x, int y) {
41          if (hit == true) {
42              enemyMap[x][y] = 1;
43              destroyedX.add(x);
44              destroyedY.add(y);
45              shipWasHit = true;
46          } else {
47              enemyMap[x][y] = -1;
48          }
49          if (sank == true) {
50              negateSurroundingAreaAfterSinking();
51              shipWasHit = false;
52          }
53      }
54  
55      public int[][] getEnemyMap() {
56          return enemyMap;
57      }
58  
59      private void acceptableCoordinates() {
60          if (shipWasHit) {
61              for (int i = 0; i < destroyedX.size(); i++) {
62                  int x = destroyedX.get(i);
63                  int y = destroyedY.get(i);
64                  acceptOnlySurroundings(x, y);
65              }
66          } else {
67              acceptAllFreeCoordinates();
68  
69          }
70      }
71  
72      private void acceptOnlySurroundings(int x, int y) {
73          for (int i = y - 1; i <= y + 1; i++) {
74              for (int j = x - 1; j <= x + 1; j++) {
75                  if ((i == y || x == j)
76                          && i >= 0 && j >= 0 && i < enemyMap.length && j < enemyMap.length
77                          && enemyMap[j][i] == 0) {
78                      acceptableX.add(j);
79                      acceptableY.add(i);
80                  }
81              }
82          }
83      }
84  
85      private void acceptAllFreeCoordinates() {
86          for (int y = 0; y < enemyMap.length; y++) {
87              for (int x = 0; x < enemyMap.length; x++) {
88                  if (enemyMap[x][y] == 0) {
89                      acceptableX.add(x);
90                      acceptableY.add(y);
91                  }
92              }
93          }
94      }
95  
96      private void negateSurroundingAreaAfterSinking() {
97          for (int i = 0; i < destroyedX.size(); i++) {
98              for (int y = destroyedY.get(i) - 1; y <= destroyedY.get(i) + 1; y++) {
99                  for (int x = destroyedX.get(i) - 1; x <= destroyedX.get(i) + 1; x++) {
100                     if (x >= 0 && y >= 0 && x < enemyMap.length && y < enemyMap.length
101                             && enemyMap[x][y] == 0) {
102                         enemyMap[x][y] = -1;
103                     }
104                 }
105             }
106         }
107         destroyedX.clear();
108         destroyedY.clear();
109     }
