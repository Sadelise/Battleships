5   /**
6    * Class contains methods for managing the players fleet and status.
7    */
8   public abstract class Player {
9   
10      private ArrayList<Ship> ships;
11      private Ship[][] locations;
12      private int[][] enemyMap;
13  
14      public Player() {
15          ships = new ArrayList<>();
16          locations = new Ship[10][10];
17          enemyMap = new int[10][10];
18      }
19  
20      /**
21       * Method receives the results of latest turn as parameters and updates the
22       * class variables accordingly.
23       *
24       * @param hit True if the last coordinates returned contained an enemy ship
25       * @param ship Contains the enemy ship that was sunk, otherwise null
26       * @param x Contains the x coordinate that was last played
27       * @param y Contains the y coordinate that was last played
28       *
29       */
30      public abstract void feedback(Boolean hit, Ship ship, int x, int y);
31  
32      /**
33       * Method compares the coordinates received as variables to the location map
34       * of player's fleet. If the locations contains a Ship the method will call
35       * the shoot method of that particular Ship.
36       *
37       * @see battleships.logic.Ship#shoot()
38       *
39       * @param x Contains the x coordinate the opponent guessed
40       * @param y Contains the y coordinate the opponent guessed
41       *
42       * @return Ship Object that was hit or null
43       */
44      public Ship shoot(int x, int y) {
45          if (locations[x][y] != null) {
46              locations[x][y].shoot();
47              return locations[x][y];
48          }
49          return null;
50      }
51  
52      /**
53       * Method returns a list containing the player fleet of Ships.
54       *
55       * @return List of Ship Objects
56       */
57      public ArrayList<Ship> getShips() {
58          return ships;
59      }
60  
61      /**
62       * Method returns the array containing the locations of the players fleet.
63       *
64       * @return Array of Ships showing the Ship locations
65       */
66      public Ship[][] getLocations() {
67          return locations;
68      }
69  
70      /**
71       * Method returns the array containing all the locations the player has
72       * guessed.
73       *
74       * @return Array of integers, marking the guessed locations
75       */
76      public int[][] getEnemyMap() {
77          return enemyMap;
78      }
79  
80      /**
81       * Method iterates through players Ships and checks if they have sunk.
82       *
83       * @return True if all the Ships have sunk, otherwise false
84       */
85      public Boolean didPlayerLose() {
86          if (!ships.isEmpty()) {
87              for (Ship laiva : ships) {
88                  if (!laiva.didItSink()) {
89                      return false;
90                  }
91              }
92          }
93          return true;
94      }
95  
96      /**
97       * Method adds the parameter Ship to the list of Ships and adds its location
98       * information to the map after making sure the Ship is valid.
99       *
100      * @param ship The ship intended to be added to the players fleet
101      *
102      * @return True if the ship was added, otherwise false
103      */
104     public Boolean addShip(Ship ship) {
105         int[] x = ship.getXcoordinates();
106         int[] y = ship.getYcoordinates();
107         if (validateShipPosition(ship, x, y)
108                 && ship.getSize() == x.length
109                 && x.length != 0
110                 && shipPiecesAttached(x, y)) {
111             for (int i = 0; i < x.length; i++) {
112                 locations[x[i]][y[i]] = ship;
113             }
114             ships.add(ship);
115             return true;
116         } else {
117             return false;
118         }
119     }
120 
121     private Boolean validateShipPosition(Ship ship, int[] x, int[] y) {
122         for (int i = 0; i < x.length; i++) {
123             if (!isSingleCoordinateAcceptable(x[i], y[i])) {
124                 return false;
125             }
126         }
127         return true;
128     }
129 
130     private Boolean isSingleCoordinateAcceptable(int x, int y) {
131         for (int i = y - 1; i <= y + 1; i++) {
132             for (int j = x - 1; j <= x + 1; j++) {
133                 if (i >= 0 && i < locations.length && j >= 0 && j < locations.length) {
134                     if (locations[j][i] != null
135                             || !(x >= 0 && x < locations.length && y >= 0 && y < locations.length)) {
136                         return false;
137                     }
138                 }
139             }
140         }
141         return true;
142     }
143 
144     private Boolean shipPiecesAttached(int[] x, int[] y) {
145         if (x.length > 1) {
146             int changes[];
147             int stays[];
148             int staysAt;
149             if (y[0] == y[1]) {
150                 staysAt = y[0];
151                 stays = y;
152                 changes = x;
153             } else {
154                 staysAt = x[0];
155                 stays = x;
156                 changes = y;
157             }
158             int prev = changes[0];
159             for (int i = 1; i < x.length; i++) {
160                 if ((prev + 1 != changes[i] || stays[i] != staysAt)) {
161                     return false;
162                 }
163                 prev = changes[i];
164             }
165         }
166         return true;
167     }

169     /**
170      * Method automatically changes the values of squares on enemy map adjacent
171      * to a sunken ship to -1.
172      *
173      * @param ship The sunken Ship
174      */
175     public void negateSurroundingAreaAfterSinking(Ship ship) {
176         for (int i = 0; i < ship.getXcoordinates().length; i++) {
177             for (int y = ship.getYcoordinates()[i] - 1; y <= ship.getYcoordinates()[i] + 1; y++) {
178                 for (int x = ship.getXcoordinates()[i] - 1; x <= ship.getXcoordinates()[i] + 1; x++) {
179                     if (x >= 0 && y >= 0 && x < enemyMap.length && y < enemyMap.length
180                             && enemyMap[x][y] == 0) {
181                         enemyMap[x][y] = -1;
182                     }
183                 }
184             }
185         }
186     }
