1   package battleships.logic;
2   
3   import battleships.domain.Ship;
4   
5   /**
6    * Class contains methods for ensuring that player's ships are correctly shaped
7    * and laid out according to rules.
8    */
9   public class ShipValidator {
10  
11      /**
12       * Method makes sure ship's coordinates are correctly placed.
13       *
14       * @param x x coordinate to be checked
15       * @param y y coordinate to be checked
16       * @param locations the locations of player's existing ships
17       * @return True if coordinates are placed according to restrictions
18       * otherwise false.
19       */
20      public Boolean validateShip(int[] x, int[] y, Ship[][] locations) {
21          if (!shipPiecesAttached(x, y)) {
22              return false;
23          }
24          for (int i = 0; i < x.length; i++) {
25              if (!isSingleCoordinateAcceptable(x[i], y[i], locations)) {
26                  return false;
27              }
28          }
29          return true;
30      }
31  
32      private Boolean isSingleCoordinateAcceptable(int x, int y, Ship[][] locations) {
33          for (int i = y - 1; i <= y + 1; i++) {
34              for (int j = x - 1; j <= x + 1; j++) {
35                  if (withinBoundaries(j, i, locations.length)) {
36                      if (locations[j][i] != null || !(withinBoundaries(x, y, locations.length))) {
37                          return false;
38                      }
39                  }
40              }
41          }
42          return true;
43      }
44  
45      private Boolean shipPiecesAttached(int[] x, int[] y) {
46          if (x.length > 1) {
47              int changes[], stays[], staysAt;
48              if (y[0] == y[1]) {
49                  staysAt = y[0];
50                  stays = y;
51                  changes = x;
52              } else {
53                  staysAt = x[0];
54                  stays = x;
55                  changes = y;
56              }
57              for (int i = 1; i < x.length; i++) {
58                  if ((changes[i - 1] + 1 != changes[i] || stays[i] != staysAt)) {
59                      return false;
60                  }
61              }
62          }
63          return true;
64      }
65  
66      /**
67       * Method makes sure given coordinates are within the dimensions of the game
68       * area.
69       *
70       * @param x x coordinate to be checked
71       * @param y y coordinate to be checked
72       * @param length the length of the game area
73       * @return True if coordinates within limits, otherwise false
74       */
75      public Boolean withinBoundaries(int x, int y, int length) {
76          return y >= 0
77                  && x >= 0
78                  && y < length
79                  && x < length;
80      }
81  }
