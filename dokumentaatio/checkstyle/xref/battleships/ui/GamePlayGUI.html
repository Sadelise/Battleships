1   package battleships.ui;
2   
3   import battleships.logic.Battleships;
4   import java.awt.Color;
5   import java.awt.Dimension;
6   import javax.swing.JButton;
7   import javax.swing.JLabel;
8   import javax.swing.JPanel;
9   import net.miginfocom.swing.MigLayout;
10  
11  public class GamePlayGUI extends GUI {
12  
13      private final ControlGUI main;
14      private final JButton[][] player1ButtonMap;
15      private final JButton[][] player2ButtonMap;
16  
17      public GamePlayGUI(ControlGUI main) {
18          this.main = main;
19          Battleships game = main.getGame();
20          this.player1ButtonMap = new JButton[game.getPlayer1().getLocations().length][game.getPlayer1().getLocations().length];
21          this.player2ButtonMap = new JButton[game.getPlayer2().getLocations().length][game.getPlayer2().getLocations().length];
22      }
23  
24      @Override
25      public JPanel getPanel() {
26          JPanel panel = new JPanel(new MigLayout(
27                  "fill",
28                  "[]20[]",
29                  "[]20[]"));
30          panel.setBackground(Color.white);
31          JLabel player1 = new JLabel("Player 1's turn");
32          player1.setForeground(Color.red);
33          JLabel player2 = new JLabel("Player 2");
34          JLabel winner = new JLabel(" ");
35          JPanel player1Board = board(player1ButtonMap);
36          JPanel player2Board = board(player2ButtonMap);
37          JButton newGame = new JButton("New Game");
38          JButton quit = new JButton("Quit");
39          newGame.setVisible(false);
40          quit.setVisible(false);
41  
42          panel.add(player1, "center");
43          panel.add(player2, "center, wrap");
44          panel.add(player1Board, "h 350!, w 350!, center");
45          panel.add(player2Board, "h 350!, w 350!, center, wrap");
46          panel.add(winner, "span, center, wrap");
47          panel.add(newGame, "center");
48          panel.add(quit, "center, wrap");
49  
50          GamePlayListener listener = new GamePlayListener(main, player1ButtonMap, player2ButtonMap, player1, player2, winner, newGame, quit);
51          addActionListener(listener, player1ButtonMap);
52          addActionListener(listener, player2ButtonMap);
53          newGame.addActionListener(listener);
54          quit.addActionListener(listener);
55  
56          return panel;
57      }
58  
59      public JPanel board(JButton[][] buttonMap) {
60          JPanel panel = new JPanel(new MigLayout(
61                  "",
62                  "[]0[]",
63                  "[]0[]"));
64          panel.setBackground(Color.white);
65  
66          for (int y = 0; y < 10; y++) {
67              for (int x = 0; x < 10; x++) {
68                  JButton button = new JButton();
69                  button.setSize(new Dimension(100, 100));
70                  buttonMap[x][y] = button;
71                  if (x == 9) {
72                      panel.add(button, "height 100, width 25, wrap");
73                  } else {
74                      panel.add(button, "height 100, width 25");
75                  }
76              }
77          }
78  
79          return panel;
80      }
81  
82      private void addActionListener(GamePlayListener listener, JButton[][] buttonMap) {
83          for (JButton[] buttons : buttonMap) {
84              for (JButton button : buttons) {
85                  button.addActionListener(listener);
86              }
87          }
88      }
89  }
