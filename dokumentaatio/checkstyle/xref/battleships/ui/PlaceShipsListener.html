3   import battleships.logic.ShipBuilder;
4   import battleships.logic.Battleships;
5   import battleships.domain.Player;
6   import java.awt.Color;
7   import java.awt.event.ActionEvent;
8   import java.awt.event.ActionListener;
9   import java.util.Map;
10  import javax.swing.JButton;
11  import javax.swing.JLabel;
12  
13  public class PlaceShipsListener implements ActionListener {
14  
15      private final ControlGUI main;
16      private final Map<String, JButton> buttons;
17      private final JButton[][] buttonMap;
18      private final ShipBuilder sb;
19      private int size;
20      private int orientation;
21      private JButton pressed;
22      private final JLabel error;
23      private final int mode;
24      private final JLabel playerLabel;
25      private final Battleships game;
26      private Player placingShips;
27  
28      public PlaceShipsListener(ControlGUI main, Map<String, JButton> buttons, JButton[][] buttonMap, JLabel error, JLabel playerLabel) {
29          this.main = main;
30          this.buttons = buttons;
31          this.buttonMap = buttonMap;
32          sb = new ShipBuilder();
33          orientation = 0;
34          this.error = error;
35          error.setForeground(Color.red);
36          this.mode = main.getGame().getMode();
37          this.playerLabel = playerLabel;
38          this.game = main.getGame();
39          this.placingShips = game.getPlayer1();
40      }
41  
42      @Override
43      public void actionPerformed(ActionEvent e) {
44          wasShipChosen(e);
45          checkGameArea(e);
46  
47          if (e.getSource() == buttons.get("start")) {
48              if (game.hasPlayerFinishedPlacingShips(placingShips)) {
49                  if ((game.getMode() == 1 && placingShips == game.getPlayer1())
50                          || (game.getMode() == 2 && placingShips == game.getPlayer2())) {
51                      GUI play = new GamePlayGUI(main);
52                      main.switchTo(play);
53                  } else if (placingShips == game.getPlayer1()) {
54                      resetButtons();
55                      placingShips = game.getPlayer2();
56                      playerLabel.setText("Player 2");
57                  }
58              } else if ((placingShips == game.getPlayer1()
59                      && !game.hasPlayerFinishedPlacingShips(game.getPlayer1()))
60                      || (placingShips == game.getPlayer2()
61                      && (!game.hasPlayerFinishedPlacingShips(game.getPlayer2()))
62                      && !game.getPlayer2().getShips().isEmpty())) {
63                  error.setText("You must place all the available ships.");
64              }
65          }
66  
67          if (e.getSource()
68                  == buttons.get("reset")) {
69              resetButtons();
70              placingShips.resetShips();
71          }
72  
73          if (e.getSource()
74                  == buttons.get("randomize")) {
75              error.setText(" ");
76              resetButtons();
77              placingShips.resetShips();
78              game.newFleet(placingShips);
79              showPlacements(placingShips);
80          }
81      }
82  
83      private void checkGameArea(ActionEvent e) {
84          for (int i = 0; i < buttonMap.length; i++) {
85              for (int j = 0; j < buttonMap.length; j++) {
86                  if (e.getSource() == buttonMap[j][i]) {
87                      if (pressed != null) {
88                          if (main.getGame().newShip(j, i, orientation, size, placingShips)) {
89                              pressed.setEnabled(false);
90                              size = 0;
91                              pressed = null;
92                              showPlacements(placingShips);
93                              error.setText("");
94  
95                          } else {
96                              error.setText("<html><center>You may have tried to place your ship too close to another ship <br>"
97                                      + "or inside the wall. Please try again.</center></html>");
98                          }
99                      } else {
100                         error.setText("You must choose the ship you want to place before trying to place it.");
101                     }
102                 }
103             }
104         }
105     }
106 
107     private void showPlacements(Player player) {
108         for (int i = 0; i < player.getLocations().length; i++) {
109             for (int j = 0; j < player.getLocations().length; j++) {
110                 if (player.getLocations()[j][i] != null) {
111                     buttonMap[j][i].setEnabled(false);
112                     buttonMap[j][i].setBackground(Color.black);
113                 }
114             }
115         }
116     }
117 
118     private void wasShipChosen(ActionEvent e) {
119         if (e.getSource() == buttons.get("boat5")) {
120             size = 5;
121             pressed = buttons.get("boat5");
122         }
123         if (e.getSource() == buttons.get("boat4")) {
124             size = 4;
125             pressed = buttons.get("boat4");
126         }
127         if (e.getSource() == buttons.get("boat3a")) {
128             size = 3;
129             pressed = buttons.get("boat3a");
130         }
131         if (e.getSource() == buttons.get("boat3b")) {
132             size = 3;
133             pressed = buttons.get("boat3b");
134         }
135         if (e.getSource() == buttons.get("boat2")) {
136             size = 2;
137             pressed = buttons.get("boat2");
138         }
139         if (e.getSource() == buttons.get("boat1")) {
140             size = 1;
141             pressed = buttons.get("boat1");
142         }
143         if (e.getSource() == buttons.get("toggleDirection")) {
144             if (orientation == 0) {
145                 orientation = 1;
146                 buttons.get("toggleDirection").setText("vertical");
147             } else {
148                 orientation = 0;
149                 buttons.get("toggleDirection").setText("horizontal");
150             }
151         }
152     }
153 
154     private void resetButtons() {
155         for (JButton[] cbuttons : buttonMap) {
156             for (JButton b : cbuttons) {
157                 b.setEnabled(true);
158                 b.setBackground(new JButton().getBackground());
159             }
160         }
161         for (JButton button : buttons.values()) {
162             button.setEnabled(true);
163         }
164     }
165 }
