27      public JPanel getPanel() {
28          JPanel panel = new JPanel(new MigLayout(
29                  "fill",
30                  "50[]20[]100",
31                  ""));
32          panel.setBackground(Color.white);
33          JLabel playerLabel = new JLabel("Player 1", SwingConstants.CENTER);
34          JLabel instruction = new JLabel("", SwingConstants.CENTER);
35          instruction.setText("<html><body><center><b>To place your ships:</b><br>"
36                  + "- click the ship you want to place<br>"
37                  + "- click the spot where you want your ship to start from.<br> "
38                  + "- change the direction your boat is facing by pressing the toggle button.</center></html></body>");
39          instruction.setFont(new Font("Serif", Font.PLAIN, 12));
40          JLabel error = new JLabel(" ", SwingConstants.CENTER);
41          JButton start = new JButton("Start");
42          buttons.put("start", start);
43          panel.add(instruction, "skip, wrap, center");
44          panel.add(playerLabel, "skip, wrap, center");
45          panel.add(shipsPanel());
46          panel.add(coordinates(), "h 350!, w 350!, center, wrap");
47          panel.add(error, "height 200, width 500, center, skip, wrap");
48          panel.add(start, "skip, center, wrap");
49  
50          addActionListener(start, error, playerLabel);
51          return panel;
52      }

88      public JPanel coordinates() {
89          JPanel panel = new JPanel(new MigLayout(
90                  "",
91                  "[]0[]",
92                  "[]0[]"));
93          panel.setBackground(Color.white);
94          buttonMap = new JButton[10][10];
95  
96          for (int y = 0; y < 10; y++) {
97              for (int x = 0; x < 10; x++) {
98                  JButton button = new JButton();
99                  button.setSize(new Dimension(100, 100));
100                 buttonMap[x][y] = button;
101                 if (x == 9) {
102                     panel.add(button, "height 100, width 25, wrap");
103                 } else {
104                     panel.add(button, "height 100, width 25");
105                 }
106             }
107         }
108 
109         return panel;
110     }

112     private void addActionListener(JButton start, JLabel error, JLabel playerLabel) {
113         PlaceShipsListener listener = new PlaceShipsListener(main, buttons, buttonMap, error, playerLabel);
114         start.addActionListener(listener);
115         for (JButton[] shipButtons : buttonMap) {
116             for (JButton button : shipButtons) {
117                 button.addActionListener(listener);
118             }
119         }
120         for (JButton button : buttons.values()) {
121             button.addActionListener(listener);
122         }
123     }
