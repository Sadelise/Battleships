8   public class ControlGUI implements Runnable {
9   
10      private JFrame frame;
11      private final GUI menu;
12      private GUI inUse;
13      private Battleships game;
14  
15      public ControlGUI() {
16          menu = new MenuGUI(this);
17      }
18  
19      @Override
20      public void run() {
21          frame = new JFrame("Battleships");
22          frame.setPreferredSize(new Dimension(800, 550));
23          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
24  
25          switchTo(menu);
26      }
27  
28      public JFrame getFrame() {
29          return frame;
30      }
31  
32      public void switchTo(GUI opening) {
33          opening.run();
34          inUse = opening;
35          frame.getContentPane().removeAll();
36          opening.start(frame.getContentPane());
37          frame.pack();
38          frame.setVisible(true);
39      }
40  
41      public void exit() {
42          System.exit(0);
43      }
44  
45      public GUI getMenu() {
46          return this.menu;
47      }
48  
49      public void setGame(Battleships game) {
50          this.game = game;
51      }
52  
53      public Battleships getGame() {
54          if (game != null) {
55              return this.game;
56          }
57          return null;
58      }
59  }
