3   /**
4    * Class contains methods tailored for the requirements of a human player.
5    */
6   public class Person extends Player {
7   
8       /**
9        * Method receives the results of latest turn as parameters and updates the
10       * class variables accordingly.
11       *
12       * @param hit True if the last coordinates returned contained an enemy ship
13       * @param ship Contains the enemy ship that was sunk, otherwise null
14       * @param x Contains the x coordinate that was last played
15       * @param y Contains the y coordinate that was last played
16       *
17       */
18      @Override
19      public void feedback(Boolean hit, Ship ship, int x, int y) {
20          if (hit == true) {
21              super.getEnemyMap()[x][y] = 1;
22          } else {
23              super.getEnemyMap()[x][y] = -1;
24          }
25          if (ship != null) {
26              super.negateSurroundingAreaAfterSinking(ship);
27          }
28      }
29  }
